```mermaid
flowchart TD
    %% 定义样式
    classDef entity fill:#dae8fc,stroke:#333,stroke-width:2px;   % 矩形（实体）
    classDef attr fill:#fff2cc,stroke:#333,stroke-width:1px;     % 椭圆（属性）
    classDef relation fill:#d5e8d4,stroke:#333,stroke-width:2px; % 菱形（关系）

    %% 实体（矩形）
    user(["小程序用户"]):::entity
    admin(["后台管理员"]):::entity
    novel(["小说信息"]):::entity
    post(["帖子信息"]):::entity

    %% 属性（椭圆）- 仅列举关键属性
    user_id(("用户ID")):::attr
    username(("用户名")):::attr
    password(("密码")):::attr
    nickname(("昵称")):::attr
    regtime(("注册时间")):::attr

    admin_id(("管理员ID")):::attr
    admin_name(("管理员名")):::attr
    admin_pwd(("密码")):::attr

    novel_id(("小说ID")):::attr
    title(("书名")):::attr
    author(("作者")):::attr
    category(("分类")):::attr
    descr(("简介")):::attr

    post_id(("帖子ID")):::attr
    post_title(("标题")):::attr
    content(("内容")):::attr
    post_time(("发布时间")):::attr

    %% 关系的属性
    collect_time(("收藏时间")):::attr
    rate_type(("评价类型")):::attr
    rate_time(("评价时间")):::attr
    like_time(("点赞时间")):::attr

    %% 关系（菱形）
    collect[/"收藏"/]:::relation
    rate[/"评价"/]:::relation
    like[/"点赞"/]:::relation
    manage_novel[/"管理小说"/]:::relation
    manage_post[/"审核帖子"/]:::relation

    %% 属性连接到实体（为简洁，只画关键连线）
    user_id --> user
    username --> user
    password --> user
    nickname --> user
    regtime --> user

    admin_id --> admin
    admin_name --> admin
    admin_pwd --> admin

    novel_id --> novel
    title --> novel
    author --> novel
    category --> novel
    descr --> novel

    post_id --> post
    post_title --> post
    content --> post
    post_time --> post

    collect_time --> collect
    rate_type --> rate
    rate_time --> rate
    like_time --> like

    %% 多对多关系：用户-小说（收藏）
    user -- "1" --> collect
    collect -- "n" --> user
    novel -- "1" --> collect
    collect -- "m" --> novel

    %% 多对多关系：用户-小说（评价）
    user -- "1" --> rate
    rate -- "n" --> user
    novel -- "1" --> rate
    rate -- "m" --> novel

    %% 多对多关系：用户-帖子（点赞）
    user -- "1" --> like
    like -- "n" --> user
    post -- "1" --> like
    like -- "m" --> post

    %% 一对多关系：管理员-小说
    admin -- "1" --> manage_novel
    manage_novel -- "n" --> novel

    %% 一对多关系：管理员-帖子
    admin -- "1" --> manage_post
    manage_post -- "n" --> post
```
